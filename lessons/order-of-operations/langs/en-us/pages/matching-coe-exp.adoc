Cut out the cards in the table below, mix them up, and try to match the Circle of Evaluation with the Arithmetic Expression.

@do{
  (define expr1 '(+ -2 5))
  (define expr2 '(+ 10 2))
  (define expr3 '(+ 5 (* 10 -2)))
  (define expr4 '(+ (/ 10 -2) 5))
  (define expr5 '(* 2 (- -10 5)))
  (define expr6 '(* 5 (/ -10 -2)))
  (define expr7 '(/ (- 10 2) (+ 10 5)))
  (define expr8 '(- -2 (* 10 (+ 5 2))))
}
