For each Circle of Evaluation on left, write the Arithmetic expression on the right

  (define expr2 '(- (* 2 3) (+ 1 0)))
  (define expr3 '(- (+ 5 8) 12))

  (define expr5 '(+ (* (- 4 1) -2) 12))
}
