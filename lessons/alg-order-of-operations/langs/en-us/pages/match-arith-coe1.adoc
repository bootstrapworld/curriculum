Draw a line from each Circle of Evaluation on the left to the corresponding code on the right.

@do{
  (define expr1 '(/ (* 1 1) 1))
  (define expr2 '(/ 1 (* 1 1)))
  (define expr3 '(* (- 1 1) (+ 1 1)))

  (define expr5 '(* (+ 1 (- 1 1)) (+ 1 1)))
}
