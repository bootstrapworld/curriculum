             ( 40 (- 660000 (* 40 250))))
#:show-examples '((#f #f #f) (#f #f #f))
#:param-list '(n)
#:show-params? #f
#:body '(- 660000 (* minutes 250))
#:show-body? #f
#:grid-lines? #t
}
